30 template<typename T>
31 static double emplace_test(T& testitem, const std::vector<std::vector<int> >& vec, const char* p_mapname)
32 {
33  double accum = 0.0;
34  for (int k = 0; k < TEST_COUNT; k++)
35  {
36  testitem.clear();
37 
38  for (int item : vec[k])
39  {
40  auto start = std::chrono::system_clock::now();
41  testitem.emplace(item, item);
42  auto end = std::chrono::system_clock::now();
43  std::chrono::duration<double> elapsed = end - start;
44  accum += elapsed.count();
45  }
46  }
47 
48  return accum * (double)1000 / (double)TEST_COUNT;
49 
50 }

52 template<typename T>
53 static double find_test(T& testitem, const std::vector<std::vector<int> >& vec, const char* p_mapname)
54 {
55  double accum = 0.0;
56  for (int k = 0; k < TEST_COUNT; ++k)
57  {
58  testitem.clear();
59  for (int item : vec[k])
60  {
61  testitem.emplace(item, item);
62  }
63 
64  for (int item : vec[k])
65  {
66  auto start = std::chrono::system_clock::now();
67  auto result = testitem.find(item);
68  auto end = std::chrono::system_clock::now();
69  SATES_NE(result, testitem.end());
70 
71  std::chrono::duration<double> elapsed = end - start;
72  accum += elapsed.count();
73  }
74 
75  }
76  return accum * (double)1000 / (double)TEST_COUNT;
