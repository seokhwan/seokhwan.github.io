97  void reserve(uint32_t new_capacity)
98  {
99  // run if new_capacity is bigger than current capacity
100  // otherwise, do nothing
101  if (new_capacity > m_capacity)
102  {
103  // - make a new array
104  T* new_arr = new T[new_capacity];
105 
106  // - backup all data
107  for (uint32_t i = 0U; i < m_size; ++i)
108  {
109  new_arr[i] = m_p_arr[i];
110  }
111  uint32_t cursize = m_size;
112 
113  // - destroy current array
114  destroy();
115 
116  // replace it with new values
117  m_p_arr = new_arr;
118  m_capacity = new_capacity;
119  m_size = cursize;
120  }
121  }
