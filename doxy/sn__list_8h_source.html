284  {
285  SN_EXCEPTION_HANDLE((m_size >= m_capacity), LIST_EXCEPTION_PUSH_BACK_OVERFLOW, "sn_list::push_back()");
286 
287  m_size++;
288  node* p_node = m_p_vec->back();
289  p_node->reset();
290  m_p_vec->pop_back();
291 
292  p_node->element = elem;
293 
294  if (nullptr == m_p_head)

296  m_p_head = p_node;
297  m_p_tail = m_p_vec->back();

315  void pop_front()
316  {
317  SN_EXCEPTION_HANDLE((0U == m_size), LIST_EXCEPTION_POP_FRONT_NO_ELEM, "sn_list::pop_front()");
318 
319  if (1U == m_size)
320  {

322  m_p_vec->push_back(m_p_head);
323  }
324  else

326  m_p_vec->push_back(m_p_head);
327  m_p_head = m_p_head->p_next;
328  m_p_head->p_prev = nullptr;
329  }

336 
337  if (1U == m_size)

341  }
342  else

345  p_last->p_prev->p_next = m_p_tail;
346  m_p_tail->p_prev = p_last->p_prev;
347  m_p_vec->push_back(p_last);
348  }
